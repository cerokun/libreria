 24:  * Converts HTML to formatted plain text
 25:  */
 26: class Html2Text

 28:     /**
 29:      * Contains the HTML content to convert.
 30:      *
 31:      * @type string
 32:      */
 33:     protected $html;
 34: 
 35:     /**
 36:      * Contains the converted, formatted text.
 37:      *
 38:      * @type string
 39:      */
 40:     protected $text;
 41: 
 42:     /**
 43:      * Maximum width of the formatted text, in columns.
 44:      *
 45:      * Set this value to 0 (or less) to ignore word wrapping
 46:      * and not constrain text to a fixed-width column.
 47:      *
 48:      * @type integer
 49:      */
 50:     protected $width = 70;
 51: 
 52:     /**
 53:      * List of preg* regular expression patterns to search for,
 54:      * used in conjunction with $replace.
 55:      *
 56:      * @type array
 57:      * @see $replace
 58:      */
 59:     protected $search = array(
 60:         "/\r/",                                  // Non-legal carriage return
 61:         "/[\n\t]+/",                             // Newlines and tabs
 62:         '/<head[^>]*>.*?<\/head>/i',             // <head>
 63:         '/<script[^>]*>.*?<\/script>/i',         // <script>s -- which strip_tags supposedly has problems with
 64:         '/<style[^>]*>.*?<\/style>/i',           // <style>s -- which strip_tags supposedly has problems with
 65:         '/<p[^>]*>/i',                           // <P>
 66:         '/<br[^>]*>/i',                          // <br>
 67:         '/<i[^>]*>(.*?)<\/i>/i',                 // <i>
 68:         '/<em[^>]*>(.*?)<\/em>/i',               // <em>
 69:         '/(<ul[^>]*>|<\/ul>)/i',                 // <ul> and </ul>
 70:         '/(<ol[^>]*>|<\/ol>)/i',                 // <ol> and </ol>
 71:         '/(<dl[^>]*>|<\/dl>)/i',                 // <dl> and </dl>
 72:         '/<li[^>]*>(.*?)<\/li>/i',               // <li> and </li>
 73:         '/<dd[^>]*>(.*?)<\/dd>/i',               // <dd> and </dd>
 74:         '/<dt[^>]*>(.*?)<\/dt>/i',               // <dt> and </dt>
 75:         '/<li[^>]*>/i',                          // <li>
 76:         '/<hr[^>]*>/i',                          // <hr>
 77:         '/<div[^>]*>/i',                         // <div>
 78:         '/(<table[^>]*>|<\/table>)/i',           // <table> and </table>
 79:         '/(<tr[^>]*>|<\/tr>)/i',                 // <tr> and </tr>
 80:         '/<td[^>]*>(.*?)<\/td>/i',               // <td> and </td>
 81:         '/<span class="_html2text_ignore">.+?<\/span>/i'  // <span class="_html2text_ignore">...</span>
 82:     );
 83: 
 84:     /**
 85:      * List of pattern replacements corresponding to patterns searched.
 86:      *
 87:      * @type array
 88:      * @see $search
 89:      */
 90:     protected $replace = array(
 91:         '',                                     // Non-legal carriage return
 92:         ' ',                                    // Newlines and tabs
 93:         '',                                     // <head>
 94:         '',                                     // <script>s -- which strip_tags supposedly has problems with
 95:         '',                                     // <style>s -- which strip_tags supposedly has problems with
 96:         "\n\n",                                 // <P>
 97:         "\n",                                   // <br>
 98:         '_\\1_',                                // <i>
 99:         '_\\1_',                                // <em>
100:         "\n\n",                                 // <ul> and </ul>
101:         "\n\n",                                 // <ol> and </ol>
102:         "\n\n",                                 // <dl> and </dl>
103:         "\t* \\1\n",                            // <li> and </li>
104:         " \\1\n",                               // <dd> and </dd>
105:         "\t* \\1",                              // <dt> and </dt>
106:         "\n\t* ",                               // <li>

396:     /**
397:      * Workhorse function that does actual conversion.
398:      *
399:      * First performs custom tag replacement specified by $search and
400:      * $replace arrays. Then strips any remaining HTML tags, reduces whitespace
401:      * and newlines to a readable format, and word wraps the text to
402:      * $this->_options['width'] characters.
403:      *
404:      * @param string $text Reference to HTML content string
405:      */
406:     protected function _converter(&$text)
407:     {
408:         // Convert <BLOCKQUOTE> (before PRE!)
409:         $this->_convert_blockquotes($text);
410: 
411:         // Convert <PRE>
412:         $this->_convert_pre($text);
413: 
414:         // Run our defined tags search-and-replace
415:         $text = preg_replace($this->search, $this->replace, $text);
416: 
417:         // Run our defined tags search-and-replace with callback
418:         $text = preg_replace_callback($this->callback_search, array($this, '_preg_callback'), $text);
419: 
420:         // Strip any other HTML tags
421:         $text = strip_tags($text, $this->allowed_tags);
422: 
423:         // Run our defined entities/characters search-and-replace
424:         $text = preg_replace($this->ent_search, $this->ent_replace, $text);
425: 
426:         // Replace known html entities
427:         $text = html_entity_decode($text, ENT_QUOTES);
428: 
429:         // Remove unknown/unhandled entities (this cannot be done in search-and-replace block)
430:         $text = preg_replace('/&([a-zA-Z0-9]{2,6}|#[0-9]{2,4});/', '', $text);
431: 
432:         // Convert "|+|amp|+|" into "&", need to be done after handling of unknown entities
433:         // This properly handles situation of "&amp;quot;" in input string
434:         $text = str_replace('|+|amp|+|', '&', $text);
435: 
436:         // Bring down number of empty lines to 2 max
437:         $text = preg_replace("/\n\s+\n/", "\n\n", $text);
438:         $text = preg_replace("/[\n]{3,}/", "\n\n", $text);
439: 
440:         // remove leading empty lines (can be produced by eg. P tag on the beginning)
441:         $text = ltrim($text, "\n");
442: 
443:         // Wrap the text to a readable format
444:         // for PHP versions >= 4.0.2. Default width is 75
445:         // If width is 0 or less, don't wrap the text.
446:         if ($this->_options['width'] > 0) {
447:             $text = wordwrap($text, $this->_options['width']);
448:         }

452:      * Helper function called by preg_replace() on link replacement.
453:      *
454:      * Maintains an internal list of links to be displayed at the end of the
455:      * text, with numeric indices to the original point in the text they
456:      * appeared. Also makes an effort at identifying and handling absolute
457:      * and relative links.
458:      *
459:      * @param string $link URL of the link
460:      * @param string $display Part of the text to associate number with
461:      * @param null $link_override
462:      * @return string

464:     protected function _build_link_list($link, $display, $link_override = null)
465:     {
466:         $link_method = ($link_override) ? $link_override : $this->_options['do_links'];
467:         if ($link_method == 'none') {
468:             return $display;

472:         // Ignored link types
473:         if (preg_match('!^(javascript:|mailto:|#)!i', $link)) {
474:             return $display;

540:      * Helper function for BLOCKQUOTE body conversion.
541:      *
542:      * @param string $text HTML content
543:      */
544:     protected function _convert_blockquotes(&$text)
545:     {
546:         if (preg_match_all('/<\/*blockquote[^>]*>/i', $text, $matches, PREG_OFFSET_CAPTURE)) {
547:             $start = 0;
548:             $taglen = 0;
549:             $level = 0;
550:             $diff = 0;
551:             foreach ($matches[0] as $m) {
552:                 if ($m[0][0] == '<' && $m[0][1] == '/') {
553:                     $level--;
554:                     if ($level < 0) {
555:                         $level = 0; // malformed HTML: go to next blockquote
556:                     } elseif ($level > 0) {
557:                         // skip inner blockquote
558:                     } else {
559:                         $end = $m[1];
560:                         $len = $end - $taglen - $start;
561:                         // Get blockquote content
562:                         $body = substr($text, $start + $taglen - $diff, $len);
563: 
564:                         // Set text width
565:                         $p_width = $this->_options['width'];
566:                         if ($this->_options['width'] > 0) $this->_options['width'] -= 2;
567:                         // Convert blockquote content

572:                         $body = '<pre>' . htmlspecialchars($body) . '</pre>';
573:                         // Re-set text width
574:                         $this->_options['width'] = $p_width;
575:                         // Replace content
576:                         $text = substr($text, 0, $start - $diff)
577:                             . $body . substr($text, $end + strlen($m[0]) - $diff);
578: 
579:                         $diff = $len + $taglen + strlen($m[0]) - strlen($body);
580:                         unset($body);
581:                     }
582:                 } else {
583:                     if ($level == 0) {
584:                         $start = $m[1];
585:                         $taglen = strlen($m[0]);
586:                     }
587:                     $level++;
588:                 }
